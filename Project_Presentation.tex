\section{Introduction \& Problem Statement}

\begin{frame}{The Problem: Food Insecurity through Price Volatility}
\begin{columns}
\begin{column}{0.55\textwidth}
\begin{itemize}
    \item Agriculture = \textbf{24\% of Uganda's GDP}, employs \textbf{70\%+} of workforce
    \item Staple crops: \textbf{Maize, Beans, Cassava}
    \item Prices fluctuate \textbf{30--50\%} between seasons
    \item Smallholder farmers sell post-harvest at \textbf{lowest prices}
    \item Existing WFP/FEWS NET systems are \textbf{retrospective}, not predictive
\end{itemize}
\vspace{0.5em}
\begin{block}{Our Solution}
Build a machine learning system that forecasts crop prices \textbf{4--8 weeks ahead}
\end{block}
\end{column}
\begin{column}{0.42\textwidth}
\begin{block}{Impact}
\begin{itemize}
    \item[\checkmark] Farmers optimise selling timing
    \item[\checkmark] NGOs pre-position food aid
    \item[\checkmark] Policymakers plan interventions
\end{itemize}
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Research Question \& Objectives}
\begin{block}{Research Question}
\textit{How accurately can machine learning models forecast staple crop market prices 4--8 weeks ahead in Uganda, and which model -- ARIMA, Prophet, or LSTM -- achieves the best performance when rainfall is incorporated as a predictor?}
\end{block}

\vspace{0.5em}
\textbf{Five Objectives:}
\begin{enumerate}
    \item \textbf{Collect \& Merge} WFP prices + CHIRPS rainfall + World Bank commodity indices
    \item \textbf{Engineer Features:} lags, rolling stats, seasonal encoding, climate covariates
    \item \textbf{Implement 3 Models:} ARIMA, Facebook Prophet, LSTM
    \item \textbf{Evaluate \& Compare} using MAE, RMSE, MAPE (80/20 chronological split)
    \item \textbf{Deploy} best model in interactive Streamlit dashboard
\end{enumerate}
\end{frame}

\section{Feature Engineering}

\begin{frame}{Transforming Raw Data into AI-Ready Features}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{1. Price Lag Features}
$$\text{price\_lag}_k(t) = \text{price}(t - k)$$
$k \in \{1, 2, 4, 8\}$ -- captures momentum \& trends
\end{block}

\begin{block}{2. Rolling Statistics}
$$\text{roll\_mean}_w(t) = \frac{1}{w}\sum_{i=0}^{w-1} \text{price}(t-i)$$
Windows: 4 months, 8 months\\
Also: rolling std.\ dev.\ (measures volatility)
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{3. Sinusoidal Seasonal Encoding}
$$\text{month\_sin} = \sin\left(\frac{2\pi m}{12}\right)$$
$$\text{month\_cos} = \cos\left(\frac{2\pi m}{12}\right)$$
Preserves cyclical continuity (Dec $\leftrightarrow$ Jan)
\end{block}

\begin{block}{4. Climate Lags}
$$\text{rain\_lag}_k(t) = \text{rainfall}(t - k)$$
$k \in \{1, 4, 8\}$ -- rainfall today affects prices in 4 months (growing cycle)
\end{block}
\end{column}
\end{columns}

\vspace{0.5em}
\centering\textbf{5. Global Covariates:} World Bank crude oil + maize indices (transport cost proxies)
\end{frame}

\section{Model Architectures}

\begin{frame}{Model 1: ARIMA -- Classical Statistical Baseline}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{ARIMA(p, d, q):}
$$y'_t = c + \sum_{i=1}^{p}\phi_i y'_{t-i} + \sum_{j=1}^{q}\theta_j \varepsilon_{t-j} + \varepsilon_t$$

\begin{itemize}
    \item \textbf{AR($p$):} Linear combination of past values
    \item \textbf{I($d$):} Differencing for stationarity
    \item \textbf{MA($q$):} Past forecast errors for self-correction
\end{itemize}

\textbf{Order Selection:} Exhaustive AIC grid search
$$\text{AIC} = 2k - 2\ln(\hat{L})$$
$p \in \{0..3\}$, $d \in \{0,1\}$, $q \in \{0..3\}$ = 32 candidates/pair
\end{column}
\begin{column}{0.4\textwidth}
\begin{block}{Strengths}
\begin{itemize}
    \item[\textcolor{lstmGreen}{\checkmark}] Interpretable coefficients
    \item[\textcolor{lstmGreen}{\checkmark}] No overfitting risk (few params)
    \item[\textcolor{lstmGreen}{\checkmark}] Works well on stationary data
\end{itemize}
\end{block}
\begin{alertblock}{Limitations}
\begin{itemize}
    \item[\textcolor{prophetRed}{$\times$}] Linear assumptions
    \item[\textcolor{prophetRed}{$\times$}] Univariate only
    \item[\textcolor{prophetRed}{$\times$}] Struggles with regime changes
\end{itemize}
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Model 2: Facebook Prophet -- Additive Decomposition}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{Prophet Decomposition:}
$$y(t) = g(t) + s(t) + \beta_{\text{rain}} \cdot \text{rainfall}(t) + \varepsilon_t$$

\begin{itemize}
    \item $g(t)$ = \textbf{Piecewise linear trend} with automatic changepoint detection
    \item $s(t)$ = \textbf{Fourier seasonal component}:
    $$s(t) = \sum_{n=1}^{N}\left(a_n\cos\frac{2\pi nt}{P} + b_n\sin\frac{2\pi nt}{P}\right)$$
    \item $\beta_{\text{rain}}$ = Learned rainfall coefficient
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\begin{block}{Strengths}
\begin{itemize}
    \item[\textcolor{lstmGreen}{\checkmark}] Handles missing data
    \item[\textcolor{lstmGreen}{\checkmark}] Strong yearly seasonality
    \item[\textcolor{lstmGreen}{\checkmark}] External regressors
\end{itemize}
\end{block}
\begin{alertblock}{Limitations}
\begin{itemize}
    \item[\textcolor{prophetRed}{$\times$}] Fixed Fourier seasonality
    \item[\textcolor{prophetRed}{$\times$}] Spurious changepoints
    \item[\textcolor{prophetRed}{$\times$}] Designed for business data
\end{itemize}
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Model 3: LSTM -- Deep Learning Neural Network}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{LSTM Gating Mechanism:}
\begin{align*}
f_t &= \sigma(W_f \cdot [h_{t-1}, x_t] + b_f) & \text{Forget Gate}\\
i_t &= \sigma(W_i \cdot [h_{t-1}, x_t] + b_i) & \text{Input Gate}\\
o_t &= \sigma(W_o \cdot [h_{t-1}, x_t] + b_o) & \text{Output Gate}\\
C_t &= f_t \odot C_{t-1} + i_t \odot \tilde{C}_t & \text{Cell State}
\end{align*}

\textbf{Cell state = ``memory highway''} carries information across time steps without degradation

\vspace{0.3em}
Solves \textbf{vanishing gradient problem} of standard RNNs
\end{column}
\begin{column}{0.4\textwidth}
\begin{block}{Our Architecture}
\small
\begin{tabular}{ll}
Layer 1 & LSTM (64 units) \\
Dropout & 20\% \\
Layer 2 & LSTM (32 units) \\
Dropout & 20\% \\
Output & Dense (1 unit) \\
\midrule
Window & 6 months \\
Features & 9 inputs \\
Optimiser & Adam \\
Loss & MSE \\
Epochs & 100 (early stop) \\
\end{tabular}
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{LSTM: Multivariate Sliding Window Approach}
\begin{block}{9-Feature Input Vector (per time step)}
\centering
\begin{tabular}{ll}
\texttt{price} & Current price (UGX) \\
\texttt{rainfall} & Monthly rainfall (mm) \\
\texttt{wb\_crude\_oil} & Global oil index \\
\texttt{wb\_maize} & Global maize index \\
\texttt{price\_lag\_1} & Price 1 month ago \\
\texttt{price\_lag\_2} & Price 2 months ago \\
\texttt{month\_sin} & Seasonal indicator \\
\texttt{month\_cos} & Seasonal indicator \\
\texttt{rain\_lag\_1} & Rainfall 1 month ago \\
\end{tabular}
\end{block}

\vspace{0.5em}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{Sliding Window}
Examines \textbf{6 consecutive months} of all 9 features to predict \textbf{month 7's price}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Normalisation}
MinMaxScaler: $x_{\text{scaled}} = \frac{x - x_{\min}}{x_{\max} - x_{\min}}$\\
Ensures all features contribute equally
\end{block}
\end{column}
\end{columns}
\end{frame}

\section{Results \& Evaluation}

\begin{frame}{Evaluation Metrics}
\begin{columns}
\begin{column}{0.33\textwidth}
\begin{block}{MAE}
$$\frac{1}{n}\sum|y_i - \hat{y}_i|$$
\centering Average error\\in UGX\\
\textit{Most interpretable}
\end{block}
\end{column}
\begin{column}{0.33\textwidth}
\begin{block}{RMSE}
$$\sqrt{\frac{1}{n}\sum(y_i - \hat{y}_i)^2}$$
\centering Penalises large\\errors more\\
\textit{Sensitive to outliers}
\end{block}
\end{column}
\begin{column}{0.33\textwidth}
\begin{block}{MAPE}
$$\frac{100\%}{n}\sum\left|\frac{y_i - \hat{y}_i}{y_i}\right|$$
\centering Scale-independent\\percentage\\
\textit{Cross-crop comparison}
\end{block}
\end{column}
\end{columns}

\vspace{1em}
\centering
\textbf{Evaluation Protocol:} Strict chronological \textbf{80/20 split}\\
No shuffling $\rightarrow$ No data leakage $\rightarrow$ No future information in training
\end{frame}

\begin{frame}{Aggregate Results: LSTM Dominates}
\begin{table}

\section{Challenges \& Limitations}

\begin{frame}{Challenges Faced \& Overcome}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{1. Data Sparsity}
\textbf{Problem:} Neural networks need 10,000+ rows; we had $\sim$200\\
\textbf{Solution:}
\begin{itemize}
    \item Dropout (20\%) -- implicit ensemble
    \item Early stopping (patience=10)
    \item Compact architecture (64$\rightarrow$32)
\end{itemize}
\end{block}

\begin{block}{2. Geographic Mismatch}
\textbf{Problem:} Three datasets use different spatial identifiers\\
\textbf{Solution:} Manual mapping dictionaries aligning districts to P-codes
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{3. Multi-step Forecasting}
\textbf{Problem:} Error compounds recursively\\
\textbf{Solution:} Recursive strategy with updated lag features at each step
\end{block}

\begin{alertblock}{Known Limitations}
\begin{itemize}
    \item Single train/test split (vs.\ time-series CV)
    \item No confidence intervals
    \item ARIMA used without exogenous vars
    \item Models cannot predict black swan events (e.g., COVID)
\end{itemize}
\end{alertblock}
\end{column}
\end{columns}
\end{frame}
